\section{Un poco de historia: Ozono}
\frame{
	\frametitle{Nuestra primera propuesta}
	\begin{enumerate}
		\item Pensar en el recorrido\footnote{Paper de Lombardi, Passerini y Cesario, 2007}
		\begin{itemize}
			\item Introducir los conceptos gradualmente.
			\item Arrancar por los fundamentales:
				\\\hspace{2em} objeto - mensaje - referencia - polimorfismo
			\item Postergar los accesorios: 
				\\\hspace{2em} clases - herencia - ...
		\end{itemize}

		\pause
		\medskip
		\item Construir una herramienta específica
		\begin{itemize}
			\item Lenguaje de programación
			\item Entorno de desarrollo			
		\end{itemize}
	\end{enumerate}
}

\frame{
	\frametitle{Ozono}	
	\begin{itemize}
		\item Entorno de objetos sin clases
		\item Basado en Pharo
		\begin{itemize}
		 \item Basado en imagen (no archivos)
		 \item Dinámico
		 \item Diagrama automático de objetos
		\end{itemize}
	\end{itemize}

\section{Nueva propuesta: Wollok}
\frame{
	\frametitle{Wollok the language}	
	\begin{itemize}
    \item Integra clases y objetos
    \item Sintaxis \emph{educativa}
			\begin{itemize}
				\item Énfasis en los conceptos a transmitir (ej: method, val/var).
				\item Se eliminan conceptos innecesarios
				\item \emph{Moderna}: literales, bloques, etc.
			\end{itemize}
    \item Inferencia de tipos
    \item Basado en archivos
	\end{itemize}
}

\frame{
	\frametitle{¿Por qué nuestro propio lenguaje?}
	Porque lo puedo customizar 100\%
	\pause
	\begin{itemize}
		\item Sintaxis precisa
			\begin{itemize}
				\item method keyword
				\item return obligatorio
			\end{itemize}
		\item Más fácil detección de errores
		\item APIs no sobrecargadas (ej: colecciones)
		\item Import system
		\item Framework de testing integrado
		\pause
		\item Minimizar los conceptos a aprender
			\begin{itemize}
			 \item Propiedades (vs. métodos + variables)
			 \item Constructores muy básicos
			\end{itemize}
	\end{itemize}
}

\frame{
	\frametitle{Algunos detalles}	
	\begin{itemize}
		\item Todo es un mensaje, es obligatorio escribir objeto.mensaje()
		\item Valores y variables
		\item Literales para listas y conjuntos (próximamente diccionarios)
		\item Variables pueden tener valor inicial
		\item Objetos pueden extender de clases
		\item Precedencia de operadores
		\item Sintaxis abreviada $a += 10$
		\item Manejo de excepciones "try" / "catch" / "then always"
	\end{itemize}
}

\frame{
	\frametitle{Algunas discusiones}	
	\begin{itemize}
		\item Los objetos tienen scope global
		\item Object literals $=>$ ¿forma de compartir código?
		\item El if no es un mensaje
		\item Constructores
		\item Imagen vs. archivos
		\item Mixins
	\end{itemize}
}

\frame{
	\frametitle{¿Qué debería tener un entorno de desarrollo?}	
	\begin{itemize}
		\item \emph{Autocomplete}
		\item Navegación (ej: "Ir a la implementación")
		\item Búsqueda inteligente (ej: referencias, implementaciones, usos)
		\item Wizards
		\item Refactors automáticos
		\item Quick fixes
		\item Herramientas de debugging
		\item Herramientas de trabajo en grupo
		\item Herramientas de visualización de código
	\end{itemize}
}
